  TRIGGER trg_plan_nodes_register_id_upd CREATE TRIGGER trg_plan_nodes_register_id_upd AFTER UPDATE OF NodeId ON plan_nodes WHEN NEW.NodeId <> '' BEGIN INSERT OR IGNORE INTO plan_node_ids (NodeId) VALUES (NEW.NodeId); END
